A Path with 2 vertices.

>>> from Coordinate import Coordinate
>>> line = Line(start=Coordinate(1, 2), end=Coordinate(3, 4), id=u'id', classes=u'class')
>>> line.start
(1, 2)
>>> line.end
(3, 4)
>>> print line
<line class="class" id="id" x1="1.0000" x2="3.0000" y1="2.0000" y2="4.0000" />
